Nested maps

user_profile:
  # Standard key-value pairs
  id: 12345
  username: johndoe
  email: johndoe@example.com

  # 'address' is a key whose value is another map
  address:
    street: 123 Main St
    city: Anytown
    state: CA
    zip_code: "90210"

  # 'preferences' is another nested map
  preferences:
    theme: dark
    notifications:
      email: true
      sms: false

  • id: 12345

  • username: johndoe

  • email: johndoe@example.com

  • street: 123 Main St

  • city: Anytown

  • state: CA

  • zip_code: 90210

  • theme: dark

  • email notifications: true

  • sms notifications:
